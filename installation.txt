Database:
BACKUP DATABASE: 

pg_dump -U postgres -d alofa -f server/alofa_dbdump.sql
pg_dump -U postgres -d alofa -f server/alofa_schema.sql --schema-only
pg_dump -U postgres -d alofa -f server/alofa_data.sql --data-only



RESTORE DATABASE: 
psql -U postgres -d alofa -c "DROP SCHEMA public CASCADE; CREATE SCHEMA public;"

// do the first one lang pero if mag error gawin mo nlg lahat hahahah tapos double check if nagalabas
